# Hafnia — quick command cheatsheet (web-upload workflow)
#
# Flow: собираем trainer.zip локально → грузим в дашборде через "Upload new".
# Запуск самого эксперимента — только через веб-интерфейс на
# https://hafnia.milestonesys.com/dashboard/training-aas/experiments
#
# Подробности см. trainer_instruction.txt и README.md §5.


# === Сборка trainer.zip ===

# Собрать zip из корня проекта (там где Dockerfile). Применяет .hafniaignore.
# Результат: trainer.zip рядом с исходниками.
hafnia trainer create-zip .

# Посмотреть что попало в trainer.zip (дерево файлов)
hafnia trainer view-zip trainer.zip


# === Локальная проверка перед заливкой (опционально, но рекомендуется) ===

# Собрать docker-образ ровно как платформа собирает
hafnia runc build-local trainer.zip

# Запустить тот же образ локально на sample-датасете
hafnia runc launch-local --dataset eccv-cross-city "python scripts/train.py --epochs 1"

# Альтернатива — собрать и запустить через docker вручную с симуляцией cloud:
#   DOCKER_BUILDKIT=0 docker build -t rfdetr_eccv:local .
#   docker run --rm --gpus all \
#       -e HAFNIA_CLOUD=true \
#       -v $PWD/.data/datasets/eccv-cross-city:/opt/ml/input/data/training:ro \
#       rfdetr_eccv:local python scripts/train.py --epochs 1


# === Датасеты ===

# Скачать sample-датасет локально (для локальной отладки)
hafnia dataset download eccv-cross-city --force

# Список доступных датасетов
hafnia dataset ls


# === Просмотр трейнеров и экспериментов ===

# Список моих trainer packages (после загрузки в дашборде видны и тут)
hafnia trainer ls

# Публичные трейнеры платформы (мы их не используем, но для справки)
hafnia trainer ls --visibility public

# Список запущенных / завершённых экспериментов
hafnia experiment ls

# Доступные окружения (Lite / Scale) с ценами
hafnia experiment environments


# === Конфигурация / профиль ===

hafnia configure                 # первичная настройка (API key + URL)
hafnia profile ls                # все профили
hafnia profile active            # текущий
hafnia profile use <name>        # переключиться
hafnia clear                     # удалить всю конфигурацию
