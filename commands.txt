# Hafnia — quick command cheatsheet
# Run from the trainer project root (the folder containing Dockerfile + scripts/).
# Remember: every meaningful trainer change = NEW trainer (don't `update` over old one).


# === Trainer packages ===

# А. Только залить трейнер на платформу (зип создаётся автоматически)
hafnia trainer create .

# Б. Залить + сразу запустить эксперимент одной командой
hafnia experiment create \
    --dataset eccv-cross-city \
    --trainer-path . \
    --cmd "python scripts/train.py --epochs 50"

# Собрать trainer.zip локально (если хочешь грузить через веб-портал)
hafnia trainer create-zip .

# Посмотреть что лежит в trainer.zip
hafnia trainer view-zip trainer.zip

# Список своих и публичных трейнеров
hafnia trainer ls
hafnia trainer ls --visibility public

# Запустить эксперимент на уже залитом трейнере (по id)
hafnia experiment create --dataset eccv-cross-city --trainer-id <trainer-id>


# === Выбор GPU / окружения ===
# Доступные окружения (в твоём аккаунте):
#   Lite   — 1 × NVIDIA T4 (16 GB VRAM),  4 CPU,  $0.82/h,  83 credits/h
#   Scale  — 4 × NVIDIA T4 (64 GB VRAM),  48 CPU, $5.45/h, 546 credits/h
# Тип GPU фиксирован (только T4). Scale = 4 карты, не более мощная — код должен уметь multi-GPU (DDP/accelerate).

# Посмотреть актуальный список окружений
hafnia experiment environments

# 1 × T4 — дешёво, для отладки
hafnia experiment create \
    -d eccv-cross-city \
    -p . \
    -e Lite \
    -c "python scripts/train.py --epochs 5"

# 4 × T4 — полноценная тренировка
hafnia experiment create \
    -d eccv-cross-city \
    -p . \
    -e Scale \
    -c "python scripts/train.py --epochs 50"





# === Локальная проверка перед заливкой ===

# Собрать docker-образ как на платформе
hafnia runc build-local trainer.zip

# Запустить тот же образ локально на sample-датасете
hafnia runc launch-local --dataset eccv-cross-city "python scripts/train.py"


# === Датасеты ===

# Принудительно перекачать датасет
hafnia dataset download eccv-cross-city --force

# Список доступных датасетов
hafnia dataset ls


# === Конфигурация / профиль ===

hafnia configure                 # первичная настройка (API key + URL)
hafnia profile ls                # все профили
hafnia profile active            # текущий
hafnia profile use <name>        # переключиться
hafnia clear                     # удалить всю конфигурацию
